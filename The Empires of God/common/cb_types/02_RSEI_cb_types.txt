imperial_reconquest_hre_sicily = {
	name = CB_NAME_IMPERIAL
	war_name = WAR_NAME_IMPERIAL
	sprite = 17
	truce_days = 3650
	is_permanent = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	can_use = {
		ROOT = {
			has_landed_title = e_hre
			has_global_flag = ser_compromis_huitdouze
			independent = yes
			religion_group = christian
			NOT = { same_realm = FROM }
		}
		
	}

	can_use_title = {
			de_jure_liege_or_above = k_sicily
			de_jure_liege_or_above = k_venice
	}
	
	is_valid = {
		ROOT = {
			independent = yes
			religion_group = christian
			NOT = { same_realm = FROM }
		}
	}
	
	on_success = {
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 100
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 100
			}
		}
		
		FROM = { prestige = -100 }
	}

	on_success_title = {
		custom_tooltip = { 
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
					}
				}
			}
		}
		if = {
			limit = {
				holder_scope = { character = FROM }
			}
			usurp_title_only = ROOT
		}
	}

	on_fail = {
		ROOT = { prestige = -150 }
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			ROOT = { prestige = -200 }
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_piety = 100
			participation_scaled_prestige = 200
			if = {
				limit = {
					religion_group = muslim
				}
				participation_scaled_decadence = -10
			}
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 100
				participation_scaled_prestige = 200
				if = {
					limit = {
						religion_group = muslim
					}
					participation_scaled_decadence = -10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}

relique_ruler = {
	name = CB_NAME_RELIQUE_RULER
	war_name = WAR_NAME_RELIQUE_RULER
	sprite = 11
	truce_days = 3650
	
	is_permanent = yes
	can_ask_to_join_war = no

	can_use = {
		NOT = { year = 1020 }
		ROOT = {
		religion_group = christian
		OR = {
			has_landed_title = k_france
			has_landed_title = e_hre
			has_landed_title = k_lotharingia
		     }
		}
		FROM = {
			trait = relique_charles
			NOT = { 
				dynasty = 25061
			}
		}
	}

	is_valid = {
		FROM = {
			trait = relique_charles
		}
	}

	on_success = {
		if = {
			limit = { ROOT = { has_landed_title = k_france } }
			set_global_flag = relique_char_fr
		}
		if = {
			limit = { ROOT = { has_landed_title = e_hre } }
			clr_global_flag = relique_char_fr
		}
		if = {
			limit = { ROOT = { has_landed_title = k_lotharingia } }
			clr_global_flag = relique_char_fr
		}

		ROOT = {
			prestige = 150
				piety = 50
				add_trait = relique_charles
		}

		FROM = {
			prestige = -200
				remove_trait = relique_charles
		}

		on_fail = {
			ROOT = {
				piety = -50
					prestige = -150
			}
		}

		on_reverse_demand = {
			ROOT = {
				piety = -100
					prestige = -200
					transfer_scaled_wealth = {
						to = FROM
							value = 4.0
					}
			}
			FROM = {
				prestige = 200
			}
		}


		attacker_ai_victory_worth = {
			factor = -1 # always accept
		}

		attacker_ai_defeat_worth = {
			factor = 100
		}

		defender_ai_victory_worth = {
			factor = -1 # always accept
		}

		defender_ai_defeat_worth = {
			factor = 100
		}
	}
}
